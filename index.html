import sys def run_engine(): print("=============================================") print(" TRADING STRATEGY ENGINE v2.0 ") print("=============================================") try: # Inputs mode = input("Select Platform (MT5 / ETRADE): ").strip().upper() balance = float(input("Account Balance ($): ")) risk_pct = float(input("Risk (%): ")) atr = float(input("14-Day ATR: ")) high15 = float(input("15m Bracket High: ")) low15 = float(input("15m Bracket Low: ")) except ValueError: print("❌ Error: Invalid numeric input.") return # Calculations raw_range = high15 - low15 inst_buffer = atr * 0.10 buffered_stop_distance = high15 - (low15 - inst_buffer) saturation = (raw_range / atr) * 100 cash_risk = balance * (risk_pct / 100) # Gatekeeper: 32.5% to 52.5% print("\n--- STATUS ---") if not (32.5 <= saturation <= 52.5): print(f"❌ NO TRADE: Saturation {saturation:.1f}% is outside sweet spot (32.5% - 52.5%).") return else: print(f"✅ TRADE APPROVED (Saturation: {saturation:.1f}%)") # Outputs print("\n--- EXECUTION TARGETS ---") print(f"Trigger Entry: {high15:.2f}") print(f"Stop Loss : {(low15 - inst_buffer):.2f}") print(f"Target (1:1) : {(high15 + buffered_stop_distance):.2f}") print("\n--- SPLIT ORDER EXECUTION ---") if mode == 'MT5': total_lots = cash_risk / buffered_stop_distance # 2/3 Core, 1/3 Runner core_lots = total_lots * (2/3) runner_lots = total_lots * (1/3) print(f"Total Volume : {total_lots:.2f} Lots") print(f"Order 1 (2/3): {core_lots:.2f} Lots (CORE CHUNK)") print(f"Order 2 (1/3): {runner_lots:.2f} Lots (RUNNER)") else: total_shares = int(cash_risk / buffered_stop_distance) # 2/3 Core, 1/3 Runner core_shares = int(total_shares * (2/3)) runner_shares = total_shares - core_shares print(f"Total Shares : {total_shares}") print(f"Order 1 (2/3): {core_shares} Shares (CORE CHUNK)") print(f"Order 2 (1/3): {runner_shares} Shares (RUNNER)") if __name__ == "__main__": run_engine()
