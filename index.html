import sys def calculate_lot_size(ticker, terminal_type, vs_percentage, bracket_high, bracket_low): """ Validates the asset's Volatility Saturation and prints a scannable, high-contrast order layout tailored to either MT5 or E*TRADE terminal interfaces. """ # Establish the tight structural sweet spot if not (32.5 <= vs_percentage <= 52.5): print("\n" + "!" * 45) print(f"❌ {ticker} IS DISQUALIFIED") print(f"Volatility Saturation ({vs_percentage}%) is outside the 32.5% - 52.5% sweet spot.") print("ACTION: Sitting on hands. Move to the next ticker on your watchlist.") print("!" * 45) return False # Fixed Risk Strategy Parameters risk_capital = 25.0 # Fixed $25 risk allocation per trade bracket_size = bracket_high - bracket_low entry_price = bracket_high + 0.02 stop_loss = bracket_low target_1to1 = entry_price + bracket_size # ---------------------------------------------------- # TERMINAL READOUT: METATRADER 5 (MT5) # ---------------------------------------------------- if terminal_type.lower() == 'mt5': print("\n" + "=" * 45) print(f"🚨 STATUS: {ticker} IS VALID! (VS: {vs_percentage}%)") print("=" * 45) print(f"▶️ ENTRY ORDER : Buy Stop @ {entry_price:.3f}") print(f"🛑 STOP LOSS : {stop_loss:.3f} [Risk: $25.00]") print("-" * 45) print(f"📦 TICKET 1 (1/3 Base Size) -> TAKE PROFIT: {target_1to1:.3f}") print(f"🏃 TICKET 2 (2/3 Runner Size)-> TAKE PROFIT: OPEN (Move SL to BE @ {entry_price:.3f})") print("=" * 45) # ---------------------------------------------------- # TERMINAL READOUT: E*TRADE # ---------------------------------------------------- elif terminal_type.lower() == 'etrade': # Share calculation based on the $25 risk ceiling risk_per_share = entry_price - stop_loss if risk_per_share <= 0: print("\n❌ Error: Bracket High must be greater than Bracket Low.") return False total_shares = int(risk_capital / risk_per_share) # Enforce exact 1/3 Base and 2/3 Runner distribution split base_shares = int(total_shares * (1/3)) runner_shares = total_shares - base_shares print("\n" + "=" * 45) print(f"🚨 STATUS: {ticker} IS VALID! (VS: {vs_percentage}%)") print("=" * 45) print(f"📊 RISK CAP: $25.00 | Total Allocation: {total_shares} Shares") print("-" * 45) print(f"🛒 ORDER 1: Buy Stop {base_shares} Shares @ {entry_price:.2f}") print(f"🎯 TARGET 1: Limit TP @ {target_1to1:.2f} | SL @ {stop_loss:.2f}") print("-" * 45) print(f"🛒 ORDER 2: Buy Stop {runner_shares} Shares @ {entry_price:.2f}") print(f"🎯 TARGET 2: Limit TP @ NONE (Runner) | SL @ {stop_loss:.2f}") print("=" * 45) return True def main(): """ Main interactive morning execution loop. Processes one ticker at a time to eliminate decision fatigue at market open. """ print("=============================================") print(" MORNING STRATEGIC ROTATION ENGINE v2.0 ") print("=============================================") while True: # Step 1: Select execution interface terminal = input("\nSelect Terminal (mt5 / etrade) or type 'exit': ").strip() if terminal.lower() == 'exit': print("Exiting tool. Good luck in the markets today!") break if terminal.lower() not in ['mt5', 'etrade']: print("❌ Invalid terminal selection. Please type 'mt5' or 'etrade'.") continue # Step 2: Input active ticker data sequential check symbol = input("Enter Ticker/Symbol (e.g., SOFI, XAGUSD): ").strip().upper() try: vs = float(input(f"Enter Volatility Saturation for {symbol} (%): ").strip().replace('%', '')) high = float(input(f"Enter 15-Minute Bracket High: ").strip()) low = float(input(f"Enter 15-Minute Bracket Low: ").strip()) except ValueError: print("❌ Invalid numeric input. Please restart the query for this ticker.") continue # Step 3: Run validation and print instructions trade_executed = calculate_lot_size(symbol, terminal, vs, high, low) if trade_executed: print("\n⚙️ Orders generated. Copy numbers directly into your active terminal window.") print("Closing application for the session.") break # Break the loop because a valid trade configuration was logged else: print("\n🔄 Ready for next ticker input...") if __name__ == "__main__": try: main() except KeyboardInterrupt: print("\n\nTool manually closed. Good luck trading!") sys.exit(0)
