def calculate_lot_size(ticker, terminal_type, vs_percentage, bracket_high, bracket_low): # Establish your tight structural sweet spot if not (32.5 <= vs_percentage <= 52.5): print(f"\n❌ {ticker} is DISQUALIFIED (Volatility Saturation: {vs_percentage}% is out of bounds).") print("Move to the next ticker on your watchlist.") return False # Calculate exact parameters for a valid trade risk_capital = 25.0 # Your fixed $25 risk per trade bracket_size = bracket_high - bracket_low entry_price = bracket_high + 0.02 stop_loss = bracket_low target_1to1 = entry_price + bracket_size # Simple terminal sizing rules if terminal_type.lower() == 'mt5': # Typical contract size math (e.g., Silver/XAGUSD or Forex pairs) # Lot Size = Risk / (Pip/Point Value * Distance) print(f"\n✅ {ticker} is VALID! (VS: {vs_percentage}%)") print(f"--- MT5 ORDER EXECUTION (1/3 Base, 2/3 Runner) ---") print(f"Entry Stop: {entry_price:.3f} | Stop Loss: {stop_loss:.3f}") print(f"Ticket 1 (1/3 Size) -> Take Profit: {target_1to1:.3f}") print(f"Ticket 2 (2/3 Size) -> Take Profit: OPEN (Move to BE at 1:1)") elif terminal_type.lower() == 'etrade': # Equity Share Math total_shares = int(risk_capital / (entry_price - stop_loss)) base_shares = int(total_shares * (1/3)) runner_shares = total_shares - base_shares print(f"\n✅ {ticker} is VALID! (VS: {vs_percentage}%)") print(f"--- E*TRADE ORDER EXECUTION (1/3 Base, 2/3 Runner) ---") print(f"Total Position: {total_shares} shares") print(f"Order 1: Buy Stop {base_shares} shares @ {entry_price:.2f} | Limit TP: {target_1to1:.2f} | SL: {stop_loss:.2f}") print(f"Order 2: Buy Stop {runner_shares} shares @ {entry_price:.2f} | Limit TP: NONE | SL: {stop_loss:.2f}") return True # Main Interactive Morning Loop while True: print("\n==============================") terminal = input("Select Terminal (mt5 / etrade) or 'exit': ") if terminal.lower() == 'exit': break symbol = input("Enter Ticker/Symbol (e.g., SOFI, XAGUSD): ").upper() vs = float(input(f"Enter Volatility Saturation for {symbol} (%): ")) high = float(input(f"Enter 15m Bracket High: ")) low = float(input(f"Enter 15m Bracket Low: ")) trade_taken = calculate_lot_size(symbol, terminal, vs, high, low) if trade_taken: print("\nPosition fully calculated. Execution complete for the day.") break # You found your trade! Close the script. else: print("\nCycling back for the next asset...")
